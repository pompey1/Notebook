# 查询table内是否有某一个元素
``` lua
local function Contains(tempTable, test)
    for key,value in pairs(tempTable) do
        if value == test then
            return true
        end
    end
    return false
end
```
# 深拷贝
``` lua
local function DeepCopy(srcTable)
    local newTable = {}
    for i, v in pairs(srcTable) do
        if type(v) == "table" then
            newTable[i] = DeepCopy(v)
        else
            newTable[i] = v
        end
    end
    return newTable
end
--对基础类型进行深拷贝毫无意义，都会生成一个新的内存
```
# 通过字符切割字符串
``` lua
local function SplitString(str,seq)
    local r = {}
    string.gsub(str, '[^' .. seq .. ']+', function(w)
        table.insert(r, w)
    end)
    return r
end
```
# 随机获取table的randomCount个元素
``` lua
local function GetRandomNumberTable(randomTable, randomCount)
    local tempTable = {}
    local resultTable = {}
    for _, value in pairs(randomTable) do
        table.insert(tempTable, value)
    end
    for i = 1, randomCount do
        local randomIndex = math.random(1,#tempTable)
        table.insert(resultTable, tempTable[randomIndex])
        table.remove(tempTable, randomIndex)
    end
    return resultTable
end
```
